# Upplýsingar um hvernig setja skuli upp verkefnið.

1. setja upp .env með eftirfarandi breytum:

   DATABASE_URL=

   TOKEN_SECRET=

   PORT=

   CLOUDINARY_CLOUD_NAME=

   CLOUDINARY_API_KEY=

   CLOUDINARY_API_SECRET=

3. keyra npm install
4. keyra npm run setup
5. keyra npm run dev

# Dæmi um köll í vefþjónustu m.v. test gögn.
